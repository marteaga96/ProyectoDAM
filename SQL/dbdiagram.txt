TABLE USUARIOS {
  ID_USUARIO INT [PRIMARY KEY, INCREMENT, NOT NULL]
  NOMBRE_SESION VARCHAR(20) [NOT NULL]
  NOMBRE_USUARIO VARCHAR(20) [NOT NULL]
  APELLIDOS_USUARIO VARCHAR(20) [NOT NULL]
  DEPARTAMENTO_USUARIO VARCHAR(20)
}


TABLE ROLES {
  ID_USUARIO INT [PRIMARY KEY, INCREMENT, NOT NULL]
  NOMBRE_SESION VARCHAR(20) [NOT NULL]
  ROL INT [NOT NULL]
}


TABLE PRODUCTOS {
  ID_PRODUCTO INT [PRIMARY KEY, INCREMENT, NOT NULL]
  NOMBRE_SESION VARCHAR(25) [NOT NULL]
  PRECIO_SIN_IVA_PRODUCTO DOUBLE [NOT NULL]
  IVA_PRODUCTO DOUBLE [NOT NULL]
  PRECIO_TOTAL_PRODUCTO DOUBLE [NOT NULL]
}

TABLE INVENTARIO {
  ID_PRODUCTO INT [PRIMARY KEY, INCREMENT, NOT NULL]
  NOMBRE_PRODUCTO VARCHAR(20) [NOT NULL]
  EXISTENCIAS_PRODUCTO DOUBLE [NOT NULL]
}

TABLE COMPRAS {
  ID_COMPRA INT [PRIMARY KEY, INCREMENT, NOT NULL]
  ID_PROVEEDOR INT [NOT NULL]
  APROBADO_POR VARCHAR(25) [NOT NULL]
  NOMBRE_PROVEEDOR VARCHAR(25) [NOT NULL]
  ID_PRODUCTO INT [NOT NULL]
  NOMBRE_PRODUCTO VARCHAR(25) [NOT NULL]
  CANTIDAD_COMPRADA DOUBLE [NOT NULL]
  HORA TIMESTAMP [NOT NULL]
}

TABLE VENTAS {
  ID_VENTA INT [PRIMARY KEY, INCREMENT, NOT NULL]
  ID_CLIENTE INT [NOT NULL]
  APROBADO_POR VARCHAR(25) [NOT NULL]
  NOMBRE_CLIENTE VARCHAR(25) [NOT NULL]
  ID_PRODUCTO INT [NOT NULL]
  NOMBRE_PRODUCTO VARCHAR(25) [NOT NULL]
  CANTIDAD_VENDIDA DOUBLE [NOT NULL]
  FECHA TIMESTAMP [NOT NULL]
}

TABLE PROVEEDOR {
  ID_PROVEEDOR INT [PRIMARY KEY, INCREMENT, NOT NULL]
  NOMBRE_PROVEEDOR VARCHAR(25) [NOT NULL]
  APELLIDOS_PROVEEDOR VARCHAR(25) [NOT NULL]
  DIRECCIO_PROVEEDOR VARCHAR(25) [NOT NULL]
  EMAIL_PROVEEDOR INT [NOT NULL]
  TELEFONO_PROVEEDOR INT(9) [NOT NULL]
}

TABLE CLIENTE {
  ID_CLIENTE INT [PRIMARY KEY, INCREMENT, NOT NULL]
  NOMBRE_CLEINTE VARCHAR(25) [NOT NULL]
  APELLIDOS_CLIENTE VARCHAR(25) [NOT NULL]
  DIRECCIO_CLIENTE VARCHAR(25) [NOT NULL]
  EMAIL_CLIENTE INT [NOT NULL]
  TELEFONO_CLIENTE INT(9) [NOT NULL]
}

TABLE MOVIMIENTOS {
  ID_MOVIMIENTO INT [PRIMARY KEY, INCREMENT, NOT NULL]
  COMPRA_O_VENTA VARCHAR(6) [NOT NULL]
  APROBADO_POR VARCHAR(25) [NOT NULL]
  NOMBRE_PRODUCTO VARCHAR(25) [NOT NULL]
  CANTIDAD DOUBLE [NOT NULL]
  FECHA TIMESTAMP [NOT NULL]
  
}
